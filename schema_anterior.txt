generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
  schemas  = ["agenda", "catalogos", "cursos", "seguridad", "ventas"]
}

model Cita {
  id                Int          @id @default(autoincrement())
  usuario_id        Int?
  servicio_id       Int
  empleado_id       Int? 
  fecha             DateTime
  hora              String
  estado            EstadoCita   @default(PENDIENTE)
  notas             String?
  createdAt         DateTime     @default(now())
  metodo_pago       String?
  nombre_contacto   String?
  telefono_contacto String?
  servicio          Servicio     @relation(fields: [servicio_id], references: [id])
  usuario           Usuario?     @relation(fields: [usuario_id], references: [id])
  avisos            AvisoAdmin[]

  @@map("tblcitas")
  @@schema("agenda")
}

model DiaBloquedo {
  id        Int      @id @default(autoincrement())
  fecha     DateTime @db.Date
  motivo    String?
  createdAt DateTime @default(now())

  @@map("tbldiasbloquedos")
  @@schema("agenda")
}

model HorarioDisponible {
  id        Int     @id @default(autoincrement())
  hora      String
  activo    Boolean @default(true)
  diaSemana Int     @default(1)

  @@unique([hora, diaSemana])
  @@map("tblhorarios")
  @@schema("agenda")
}

model HorarioExcepcion {

  hora   String
  motivo String?

  @@unique([fecha, hora])
  @@map("tblhorarios_excepciones")
  @@schema("agenda")
}

model HoraBloqueada {

  motivo String?

  @@unique([fecha, hora])
  @@map("tblhoras_bloqueadas")
  @@schema("agenda")
}

model Marca {
  id        Int        @id @default(autoincrement())
  nombre    String     @unique @db.VarChar(100)
  activo    Boolean    @default(true)
  productos Producto[]

  @@map("tblmarcas")
  @@schema("catalogos")
}

model Categoria {
  id        Int        @id @default(autoincrement())
  nombre    String     @unique @db.VarChar(100)
  activo    Boolean    @default(true)
  productos Producto[]

  @@map("tblcategorias")
  @@schema("catalogos")
}

model Producto {
  id                  Int             @id @default(autoincrement())
  codigo              String          @db.VarChar(50)
  nombre              String          @db.VarChar(150)
  descripcion         String?
  precio_costo        Decimal         @db.Decimal(10, 2)
  precio_venta        Decimal         @db.Decimal(10, 2)
  marca_id            Int?
  categoria_id        Int?
  stock               Int             @default(0)
  activo              Boolean?        @default(true)
  fecha_creacion      DateTime?       @default(now()) @db.Timestamp(6)
  fecha_actualizacion DateTime?       @default(now()) @db.Timestamp(6)
  imagen              String?
  detalles            DetallePedido[]
  categoria           Categoria?      @relation(fields: [categoria_id], references: [id], onDelete: NoAction, onUpdate: NoAction)
  marca               Marca?          @relation(fields: [marca_id], references: [id], onDelete: NoAction, onUpdate: NoAction)

  @@map("tblproductos")
  @@schema("ventas")
}

model Servicio {
  id          Int      @id @default(autoincrement())
  nombre      String
  descripcion String?
  precio      Float
  duracion    String
  categoria   String
  beneficios  String?  
  incluye     String?
  imagen      String?
  activo      Boolean  @default(true)
  createdAt   DateTime @default(now())
  updatedAt   DateTime

  @@map("tblservicios")
  @@schema("catalogos")

model Curso {
  id             Int       @id @default(autoincrement())
  codigo         String    @unique @db.VarChar(50)
  titulo         String    @db.VarChar(200)
  descripcion    String?
  precio_total   Decimal   @db.Decimal(10, 2)
  cupo_maximo    Int
  inscritos      Int       @default(0)
  duracion_horas Int?
  nivel          String?   @db.VarChar(50)
  activo         Boolean   @default(true)
  fecha_inicio   DateTime? @db.Date
  fecha_fin      DateTime? @db.Date
  created_at     DateTime  @default(now())
  imagenes       Json?

  @@map("tblcursos")
  @@schema("cursos")
}

model AvisoAdmin {
  id         Int      @id @default(autoincrement())
  cita_id    Int
  usuario_id Int?
  mensaje    String
  leido      Boolean  @default(false)
  createdAt  DateTime @default(now())
  cita       Cita     @relation(fields: [cita_id], references: [id])
  usuario    Usuario? @relation(fields: [usuario_id], references: [id])

  @@map("tblavisos_admin")
  @@schema("seguridad")
}

model TokenRecuperacion {
  id         Int      @id @default(autoincrement())
  usuario_id Int      @unique
  token      String   @unique
  expira     DateTime
  usado      Boolean  @default(false)
  createdAt  DateTime @default(now())
  usuario    Usuario  @relation(fields: [usuario_id], references: [id])

  @@map("tbltoken_recuperacion")
  @@schema("seguridad")
}

model Usuario {
  id                Int                @id @default(autoincrement())
  nombre            String             @db.VarChar(150)
  appaterno         String?              @db.VarChar(150)
  apmaterno         String?             @db.VarChar(150)
  correo            String             @unique @db.VarChar(100)
  password          String             @db.VarChar(255)
  telefono          String?            @db.VarChar(20)
  rol               Rol                @default(CLIENTE)
  activo            Boolean            @default(true)
  intentos_fallidos Int                @default(0)
  cuenta_bloqueada  Boolean            @default(false)
  fecha_registro    DateTime           @default(now()) @db.Timestamp(6)
  citas             Cita[]
  citasComoEmpleado Cita[]
  avisos            AvisoAdmin[]
  permisos PermisoUsuario[] 
  pedidos           Pedido[]
  tokenRecuperacion TokenRecuperacion?
  
  @@map("tblusuarios")
  @@schema("seguridad")
}

model Empleado {
  id              Int      @id @default(autoincrement())
  nombre          String   @db.VarChar(150)
  puesto          String   @db.VarChar(100)
  descripcion     String?  @db.Text
  imagen          String?  // URL de Cloudinary
  orden           Int      @default(0)   // Para ordenar en "Nosotros"
  activo          Boolean  @default(true)
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  @@map("tblempleados")
  @@schema("catalogos")   // o crea un nuevo schema si prefieres
}

model PermisoUsuario {
  id         Int     @id @default(autoincrement())
  usuario_id Int
  permiso    String  @db.VarChar(100)
  activo     Boolean @default(false)

  usuario Usuario @relation(fields: [usuario_id], references: [id], onDelete: Cascade)

  @@unique([usuario_id, permiso])
  @@map("tblpermisos_usuario")
  @@schema("seguridad")
}

model DetallePedido {
  id              Int      @id @default(autoincrement())
  pedido_id       Int
  producto_id     Int
  nombre_producto String   @db.VarChar(150)
  precio_unitario Decimal  @db.Decimal(10, 2)
  cantidad        Int
  subtotal        Decimal  @db.Decimal(10, 2)
  pedido          Pedido   @relation(fields: [pedido_id], references: [id])
  producto        Producto @relation(fields: [producto_id], references: [id])

  @@map("tbldetalle_pedidos")
  @@schema("ventas")
}

model Pedido {
  id               Int             @id @default(autoincrement())
  usuario_id       Int
  estado           EstadoPedido    @default(PENDIENTE)
  subtotal         Decimal         @db.Decimal(10, 2)
  costo_envio      Decimal         @default(0) @db.Decimal(10, 2)
  total            Decimal         @db.Decimal(10, 2)
  nombre_cliente   String          @db.VarChar(200)
  correo_cliente   String          @db.VarChar(100)
  telefono_cliente String?         @db.VarChar(20)
  fecha_pedido     DateTime        @default(now()) @db.Timestamp(6)
  detalles         DetallePedido[]
  usuario          Usuario         @relation(fields: [usuario_id], references: [id])

  @@map("tblpedidos")
  @@schema("ventas")
}

model Backup {
  id     Int       @id @default(autoincrement())
  nombre String    @db.VarChar(255)
  ruta   String
  tama_o String?   @map("tamaño") @db.VarChar(50)
  fecha  DateTime? @default(now()) @db.Timestamp(6)
  estado String?   @default("COMPLETADO") @db.VarChar(50)

  @@map("tblbackup")
  @@schema("seguridad")
}

enum EstadoCita {
  PENDIENTE
  CONFIRMADA
  COMPLETADA
  CANCELADA

  @@schema("agenda")
}

enum Rol {
  ADMIN
  DOCENTE
  EMPLEADO
  CLIENTE

  @@schema("seguridad")
}

enum EstadoPedido {
  PENDIENTE
  PAGADO
  ENVIADO
  ENTREGADO
  CANCELADO

  @@schema("ventas")
}
